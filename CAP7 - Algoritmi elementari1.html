    int suma(int n)
    {
        int s=0,c;
        while(n!=0)

            c=n%10;

            n=n/10;
        }
        return s;
    }

  • c.m.m.d.c. a 2 nr. (var.1) 

    int cmmdc(int a,int b)
    {
        while(a!=b)
            if(a>b)
                a=a-b;
            else
                b=b-a;
        return a;
    }
